21// isnan() function that isn't optimized away by fastmath, unlike std::isnan()
22[[nodiscard]] inline bool IsNanNoOpt(float f) {
23 union {
24 float f;
25 std::uint32_t x;
26 } u = {f};
27 return (u.x << 1) > 0xff000000u;
28}

206// The skew-symmetric matrix that corresponds to the
207// cross product v x (something):
208template <typename T>
209Eigen::Matrix<T, 3, 3> crossProductMatrix(const Eigen::Matrix<T, 3, 1>& v) {
210 Eigen::Matrix<T, 3, 3> result;
211 result << T(0), -v.z(), v.y(), v.z(), T(0), -v.x(), -v.y(), v.x(), T(0);
212 return result;
213}
