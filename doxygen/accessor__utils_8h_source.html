17static inline int32_t getComponentSize(const fx::gltf::Accessor::ComponentType& componentType) {
18 switch (componentType) {
19 case fx::gltf::Accessor::ComponentType::Byte:
20 case fx::gltf::Accessor::ComponentType::UnsignedByte:
21 return 1;
22 case fx::gltf::Accessor::ComponentType::Short:

26 case fx::gltf::Accessor::ComponentType::UnsignedInt:

55template <typename T>
56std::vector<T> copyAccessorBuffer(const fx::gltf::Document& model, int32_t id) {
57 const auto& accessor = model.accessors[id];
58
59 const auto tsize = getTypeSize(accessor.type);
60 const auto csize = getComponentSize(accessor.componentType);
61 const auto elsize = tsize * csize;
62
63 if (elsize != sizeof(T)) {
64 MT_LOGW(
65 "[{}] Accessor type mismatch. Expect to read {} bytes, but got type of {} bytes with {} components",
66 __func__,
67 sizeof(T),
68 csize,
69 tsize);
70 return {};
71 }
72
73 const auto& view = model.bufferViews[accessor.bufferView];
74 const auto stride = view.byteStride == 0 ? elsize : view.byteStride;
75
76 const auto& buf = model.buffers[view.buffer];
77 auto bytes = gsl::as_bytes(gsl::make_span(buf.data));
78 bytes = bytes.subspan(view.byteOffset, view.byteLength);
79 bytes = bytes.subspan(accessor.byteOffset);
80
81 std::vector<T> r(accessor.count);
82 for (size_t i = 0; i < accessor.count; i++) {
83 std::memcpy(&r[i], &bytes[i * stride], elsize);
84 }
85
86 return r;
87}

133// create accessor buffer
134template <typename T>
135int32_t createAccessorBuffer(
136 fx::gltf::Document& model,
137 gsl::span<T> data,
138 const bool align = false,
139 const bool normalized = false) {
140 // gltf 2.0 allows multiple buffers, but only the first buffer will be stored in
141 // the binary file, all other ones need to be external files. Because of this we will
142 // only ever store data in the first buffer
143
144 // check if we have a buffer
145 if (model.buffers.empty()) {
146 model.buffers.resize(1);
147 }
148
149 // create buffer for data
150 const size_t bufferIdx = 0;
151 auto& buffer = model.buffers.front();
152
153 // copy data from input buffer to output buffer
154 const size_t bufferDataStart = buffer.data.size();
155 const auto elementSize = sizeof(T);
156 auto alignedElementSize = elementSize;
157 auto dataSize = data.size() * alignedElementSize;
158
159 // check for necessary alignment
160 if (align && alignedElementSize % 4 != 0) {
161 // element is not aligned, need to get bigger element size
162 alignedElementSize = ((elementSize / 4) + 1) * 4;
163 dataSize = data.size() * alignedElementSize;
164 buffer.data.resize(bufferDataStart + dataSize);
165 // need to go over each element one by one to copy
166 for (size_t e = 0; e < data.size(); e++) {
167 std::memcpy(&buffer.data[bufferDataStart + alignedElementSize * e], &data[e], elementSize);
168 }
169 buffer.byteLength = buffer.data.size();
170 } else {
171 // no alignment needed or already aligned, just copy data in
172 buffer.data.resize(bufferDataStart + dataSize);
173 std::memcpy(&buffer.data[bufferDataStart], data.data(), dataSize);
174 buffer.byteLength = buffer.data.size();
175 }
176
177 // create bufferview
178 const size_t bufferViewIdx = model.bufferViews.size();
179 model.bufferViews.emplace_back();
180 auto& bufferView = model.bufferViews.back();
181
182 bufferView.buffer = bufferIdx;
183 bufferView.byteLength = dataSize;
184 bufferView.byteOffset = bufferDataStart;
185 if (align && elementSize % 4 != 0) {
186 bufferView.byteStride = alignedElementSize;
187 } else {
188 bufferView.byteStride = 0;
189 }
190
191 // create accessor
192 const size_t accessorIdx = model.accessors.size();
193 model.accessors.emplace_back();
194 auto& accessor = model.accessors.back();
195
196 // set component type and accessor size depending on the type
197 setAccessorType<T>(accessor);
198 accessor.bufferView = bufferViewIdx;
199 accessor.byteOffset = 0;
200 accessor.count = data.size();
201 accessor.normalized = normalized;
202
203 return accessorIdx;
204}

260// add animation
261template <typename T>
262int32_t createSampler(
263 fx::gltf::Document& model,
264 fx::gltf::Animation& anim,
265 gsl::span<T> data,
266 const gsl::span<float>& timestamps) {
267 MT_CHECK(
268 data.size() == timestamps.size(), "data: {}, timestmaps: {}", data.size(), timestamps.size());
269
270 // create new sampler
271 const int32_t index = anim.samplers.size();
272 anim.samplers.emplace_back();
273 auto& sampler = anim.samplers.back();
274 sampler.interpolation = fx::gltf::Animation::Sampler::Type::Step;
275 sampler.input = createAccessorBuffer(model, timestamps);
276 sampler.output = createAccessorBuffer(model, data);
277
278 // return sampler index
279 return index;
280}
