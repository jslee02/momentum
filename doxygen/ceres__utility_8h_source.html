34template <typename T, int N>
35Eigen::Matrix<ceres::Jet<T, N>, N, 1> buildJetVec(const Eigen::Matrix<T, N, 1>& v) {
36 Eigen::Matrix<ceres::Jet<T, N>, N, 1> result;
37 for (int k = 0; k < N; ++k) {
38 result(k).a = v(k);
39 result(k).v.setZero();
40 result(k).v(k) = 1;
41 }

46template <typename T, int Rows, int Cols>
47Eigen::Matrix<ceres::Jet<T, Cols * Rows>, Rows, Cols> buildJetMat(
48 const Eigen::Matrix<T, Rows, Cols>& m) {
49 Eigen::Matrix<ceres::Jet<T, Cols * Rows>, Rows, Cols> result;
50 for (int iRow = 0; iRow < Rows; ++iRow) {
51 for (int jCol = 0; jCol < Cols; ++jCol) {
52 result(iRow, jCol).a = m(iRow, jCol);
53 result(iRow, jCol).v.setZero();
54 result(iRow, jCol).v(iRow * Cols + jCol) = 1;
55 }
56 }
57 return result;
58}

61T times_parameterTransform_times_v(
62 const T& value,
