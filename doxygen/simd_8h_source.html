59template <size_t Alignment>
60[[nodiscard]] size_t computeOffset(const Eigen::Ref<Eigen::MatrixXf>& mat) {
61 constexpr size_t sizeOfScalar = sizeof(typename Eigen::MatrixXf::Scalar);
62 const size_t addressOffset =
63 Alignment / sizeOfScalar - (((size_t)mat.data() % Alignment) / sizeOfScalar);
64
65 // If the current alignment already meets the requirement, no offset is needed.
66 if (addressOffset == Alignment / sizeOfScalar) {
67 return 0;
68 }
69
70 return addressOffset;
71}

74template <size_t Alignment>
75void checkAlignment(const Eigen::Ref<Eigen::MatrixXf>& mat, size_t offset = 0) {
76 MT_THROW_IF(
77 (uintptr_t(mat.data() + offset)) % Alignment != 0,
78 "Matrix ({}x{}, ptr: {}) is not aligned ({}) correctly.",
79 mat.rows(),
80 mat.cols(),
81 uintptr_t(mat.data()),
82 Alignment);
83}
