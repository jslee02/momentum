92 [[nodiscard]] TransformT<T> operator*(const TransformT<T>& other) const {
93 // [ s_1*R_1 t_1 ] * [ s_2*R_2 t_2 ] = [ s_1*s_2*R_1*R_2 s_1*R_1*t_2 + t_1 ]
94 // [ 0 1 ] [ 0 1 ] [ 0 1 ]
95 const Vector3<T> trans = translation + rotation * (scale * other.translation);
96 const Quaternion<T> rot = rotation * other.rotation;
97 const T newScale = scale * other.scale;
98 return TransformT<T>(std::move(trans), std::move(rot), std::move(newScale));
99 }
