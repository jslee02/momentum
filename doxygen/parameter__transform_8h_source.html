30 inline bool operator==(const PoseConstraint& poseConstraint) const {
31 // Compare the parameterIdValue as sets
32 std::map<size_t, float> paramIdToValue1;
33 std::copy(
34 parameterIdValue.begin(),
35 parameterIdValue.end(),
36 std::inserter(paramIdToValue1, paramIdToValue1.begin()));

38 std::map<size_t, float> paramIdToValue2;

48 paramIdToValue1.begin(), paramIdToValue1.end(), paramIdToValue2.begin(), pred);
49 }

169 inline bool isApprox(const ParameterTransformT<T>& parameterTransform) const {
170 // special handling of zero sparse matrix
171 bool isTransformEqual = false;
172 if (transform.cols() > 0 && transform.rows() > 0 && parameterTransform.transform.cols() > 0 &&
173 parameterTransform.transform.rows() > 0) {
174 isTransformEqual = transform.isApprox(parameterTransform.transform);
175 } else {
176 isTransformEqual = (transform.cols() == parameterTransform.transform.cols()) &&
177 (transform.rows() == parameterTransform.transform.rows());
178 }
179 if (!isTransformEqual)
180 return false;
181
182 return (
183 (name == parameterTransform.name) &&
184 activeJointParams.isApprox(parameterTransform.activeJointParams) &&
185 (parameterSets == parameterTransform.parameterSets) &&
186 (poseConstraints == parameterTransform.poseConstraints) &&
187 (blendShapeParameters == parameterTransform.blendShapeParameters));
188 }
189};
