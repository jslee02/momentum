53// This function is designed to limit the number of times an error is logged.
54// Please note that in a multi-threaded context, its behavior may not be guaranteed.
55#define _MT_RUN_ONCE(runcode) \
56 { \
57 static std::atomic<bool> codeRan(false); \
58 if (!codeRan) { \
59 bool expected = false; \
60 if (codeRan.compare_exchange_strong(expected, true)) { \
61 runcode; \
62 } \
63 } \
64 }
