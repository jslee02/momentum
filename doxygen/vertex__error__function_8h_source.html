18template <typename T>
19struct VertexConstraintT {
20 int vertexIndex = -1;
21 T weight = 1;
22 Eigen::Vector3<T> targetPosition;
23 Eigen::Vector3<T> targetNormal;

25 template <typename T2>
26 VertexConstraintT<T2> cast() const {
27 return {
28 this->vertexIndex,
29 (T)this->weight,
30 this->targetPosition.template cast<T2>(),
31 this->targetNormal.template cast<T2>()};
32 }
33};

35enum class VertexConstraintType {
36 Position, // Target the vertex position
37 Plane, // point-to-plane distance using the target normal
38 Normal, // point-to-plane distance using the source (body) normal
39 SymmetricNormal, // Point-to-plane using a 50/50 mix of source and target normal
40};
