76 virtual double getSolverDerivatives(
77 const ModelParametersT<T>& parameters,
78 const SkeletonStateT<T>& state,
79 const size_t actualParameters,
80 Eigen::MatrixX<T>& hess,
81 Eigen::VectorX<T>& grad) {
82 const int paramSize = static_cast<int>(parameters.size());
83 double error = 0.0;
84
85 if (useHessian_) {
86 // get some ice cream, you deserve it
87 error = getGradient(parameters, state, grad);
88 getHessian(parameters, state, hess);
89 } else {
90 const int residualSize = static_cast<int>(getJacobianSize());
91
92 Eigen::MatrixX<T> jacobian = Eigen::MatrixX<T>::Zero(residualSize, paramSize);
93 Eigen::VectorX<T> residual = Eigen::VectorX<T>::Zero(residualSize);
94 int rows;
95
96 error = getJacobian(parameters, state, jacobian, residual, rows);
97
98 // Update JtJ
99 if (rows > 0) {
100 // ! In truth, on the the "actualParameters" leftmost block will be used
101 // We take advantage of this here and skip the other computations
102 const auto JtBlock2 =
103 (jacobian.topLeftCorner(rows, actualParameters).transpose() * 2.0).eval();
104
105 // Update Hessian (Hessian += 2.0 * J^T * J) using selfadjointView with rankUpdate,
106 // replacing triangularView
107 hess.template selfadjointView<Eigen::Lower>().rankUpdate(JtBlock2);
108
109 // Update JtR
110 grad.noalias() += JtBlock2 * residual.head(rows);
111 }
112 }
113
114 return error;
115 }
