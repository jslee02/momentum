19// structure for applying limits to parameter sets
20// the simplest kind of limits could be just min/max limits on parameters, but we also have the
21// option of modeling more complicated limits that depend on each other
22
23// limit type
24enum LimitType {
25 MinMax,
26 MinMaxJoint,
27 MinMaxJointPassive,
28 Linear,
29 LinearJoint,
30 Ellipsoid,
31 HalfPlane
32};
33
34[[nodiscard]] std::string_view toString(LimitType type);
35
36struct LimitMinMax {
37 size_t parameterIndex; // index of parameter influenced
38 Vector2f limits; // min and max values of the parameter
39};
40
41struct LimitMinMaxJoint {
42 size_t jointIndex; // index of joint influenced
43 size_t jointParameter; // parameter of joint influenced [ tx, ty, tz, rx, ry, rz, sc ]
44 Vector2f limits; // min and max values of the parameter
45};

47struct LimitLinear { // set joints to be similar by a linear relation i.e. p_0 = s * p_1 - o
48 size_t referenceIndex; // index of reference parameter (p_0)
49 size_t targetIndex; // index of target parameter (p_1)
50 float scale; // linear scale of parameter (x)
51 float offset; // offset (positive and negative) of acceptable parameter zone
52
53 // Range where limit is applied (in target parameter values p1). This can be used to construct
54 // piecewise linear limits. Note that the minimum value of the range is inclusive but the maximum
55 // value is noninclusive, this is to allow constructing overlapping limits without
56 // double-counting.
57 float rangeMin;
58 float rangeMax;
59};
60
61struct LimitLinearJoint {
62 size_t referenceJointIndex;
63 size_t referenceJointParameter;
64 size_t targetJointIndex;
65 size_t targetJointParameter;
66 float scale;
67 float offset;
68
69 float rangeMin;
70 float rangeMax;
71};

73struct LimitEllipsoid {

81struct LimitHalfPlane {

90union LimitData {

101 LimitData(const LimitData& rhs);

104 bool operator==(const LimitData& limitData) const;
105};

107struct ParameterLimit {
108 LimitData data; // limit data depending on the type
109 LimitType type = LimitType::MinMax; // type of limit
110 float weight = 1.0f; // limit weight
111
112 inline bool operator==(const ParameterLimit& parameterLimit) const {
113 return (
114 (data == parameterLimit.data) && (type == parameterLimit.type) &&
115 isApprox(weight, parameterLimit.weight));
116 };
117};
